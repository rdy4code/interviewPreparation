  conts arrStr = str.split(""); // passing an empty string

  return arrStr.join();

// Solution 1.1;
function reverse(str){
  return str.split("")
    .reverse()
    .join("");
}

//Logic
// Create an empty string called "reversed"
// for each character in the string
// Take the character and add into the start of "reversed"
// return the result

---
function reversedString(str) {
  let reversed = "";

  for (let charachter of str) {
    reversed = charachter + reversed;
  }
  return reversed;
}

//##### Solution 3

//Logic
// Create an array with str.split('')
// reduce helper
// return the result

---
function reversedString(str) {
   return str.split("").reduce((rev, char) => return char + rev, "");
}
